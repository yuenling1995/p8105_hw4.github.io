instacart_sub = 
  instacart %>% 
  filter(
    aisle %in% c("fresh vegetables", "fresh fruits", "packaged vegetables fruits")) %>% 
  group_by(aisle) %>% 
  count(product_name) %>% 
  mutate(rank = min_rank(desc(n))) %>% 
  filter(rank < 4) %>% 
  arrange(desc(n)) %>% 
  as_tibble()
